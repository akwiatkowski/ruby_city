  def increase_residential( amount )
    coef = Math::log( amount.to_f )
    if coef < 1.0
      coef = 1.0
    end

    cost = (amount.to_f * BUILDING_COST_RESIDENTIAL.to_f) / coef
    if @money >= cost
      @money -= cost
      @residential_capacity += amount
      return true
    else
      return false
    end

  end
